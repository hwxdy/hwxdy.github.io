class SingletonTest3{

private  SingletonTest3() {};

private static SingletonTest3 instance=null;

public static SingletonTest3 getInstance() {
	
	if(instance==null) {
		synchronized(SingletonTest3.class) {
			instance= new SingletonTest3();
			}
		}
	return instance;
	}

}
